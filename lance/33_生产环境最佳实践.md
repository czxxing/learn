## 📈 性能基准测试

### 基准测试框架

```python
import lance
import numpy as np
import time
from dataclasses import dataclass
from typing import List

@dataclass
class BenchmarkResult:
    name: str
    qps: float  # Queries Per Second
    latency_p50: float  # 毫秒
    latency_p95: float
    latency_p99: float
    throughput_mbs: float  # MB/s

class LanceBenchmark:
    def __init__(self, dataset_size=1_000_000):
        self.dataset_size = dataset_size
        self.prepare_data()
    
    def prepare_data(self):
        """准备测试数据"""
        data = {
            "id": range(self.dataset_size),
            "embedding": np.random.randn(
                self.dataset_size, 768
            ).astype(np.float32),
            "score": np.random.uniform(0, 100, self.dataset_size),
        }
        self.table = lance.write_table(data, uri="bench.lance")
        self.table.create_index(
            column="embedding",
            index_type="ivf_pq",
        )
    
    def benchmark_search(self, num_queries=1000) -> BenchmarkResult:
        """基准测试：向量搜索"""
        query = np.random.randn(768).astype(np.float32)
        
        latencies = []
        start_time = time.time()
        
        for _ in range(num_queries):
            t0 = time.time()
            _ = self.table.search(query).limit(10).to_list()
            latencies.append((time.time() - t0) * 1000)
        
        total_time = time.time() - start_time
        qps = num_queries / total_time
        
        latencies.sort()
        return BenchmarkResult(
            name="vector_search",
            qps=qps,
            latency_p50=latencies[len(latencies) // 2],
            latency_p95=latencies[int(len(latencies) * 0.95)],
            latency_p99=latencies[int(len(latencies) * 0.99)],
            throughput_mbs=qps * 768 * 4 / 1024 / 1024,
        )
    
    def benchmark_filtering(self) -> BenchmarkResult:
        """基准测试：标量过滤"""
        start = time.time()
        for _ in range(100):
            _ = self.table.search_sql(
                "SELECT * FROM table WHERE score > 50 LIMIT 1000"
            ).to_list()
        elapsed = time.time() - start
        
        qps = 100 / elapsed
        return BenchmarkResult(
            name="filtering",
            qps=qps,
            latency_p50=elapsed / 100 * 1000,
            latency_p95=elapsed / 100 * 1000,
            latency_p99=elapsed / 100 * 1000,
            throughput_mbs=0,
        )

# 运行基准测试
benchmark = LanceBenchmark(dataset_size=1_000_000)

print("=== Benchmark Results ===")
result = benchmark.benchmark_search(num_queries=1000)
print(f"Search QPS: {result.qps:.0f}")
print(f"Latency P50: {result.latency_p50:.2f}ms")
print(f"Latency P95: {result.latency_p95:.2f}ms")
print(f"Latency P99: {result.latency_p99:.2f}ms")

result = benchmark.benchmark_filtering()
print(f"Filtering QPS: {result.qps:.0f}")
```

### Docker 部署

```dockerfile
FROM python:3.11-slim

WORKDIR /app

# 安装依赖
RUN apt-get update && apt-get install -y \
    build-essential \
    && rm -rf /var/lib/apt/lists/*

RUN pip install -r requirements.txt

# 复制应用
COPY . .

# 健康检查
HEALTHCHECK --interval=30s --timeout=10s --start-period=40s --retries=3 \
    CMD python -c "import lance; lance.open('/data/lance')"

CMD ["python", "app.py"]
```

## 📊 监控告警

### Prometheus 指标

```python
from prometheus_client import Counter, Histogram, Gauge
import time

# 定义指标
request_count = Counter(
    'lance_requests_total',
    'Total requests',
    ['method', 'status']
)

request_latency = Histogram(
    'lance_request_latency_seconds',
    'Request latency'
)

active_connections = Gauge(
    'lance_active_connections',
    'Active connections'
)

# 在查询中使用
def search_with_metrics(table, query):
    with request_latency.time():
        try:
            results = table.search(query).limit(10).to_list()
            request_count.labels(method='search', status='success').inc()
            return results
        except Exception as e:
            request_count.labels(method='search', status='error').inc()
            raise
```
